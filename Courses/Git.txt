git init	inicializa un nuevo repositorio
git status	muestra el contenido que hay por adicionar al repositorio
git add "file"	adiciona un archivo especifico
git diff "file"	para mostrar que cosas estan agregadas al repositorio y cuales no 
git commit -m "descrip" hace un nuevo commit
git log		para ver todos los commits que se han generado
git show HEAD 	para ver los commits mas recientes de manera detallada
git reset "7 primeros caracteres del codigo commit"
git branch	para ver todas las branchs del repositorio
git branch "name_branch" crea una nueva branch
git checkout "name_branch" para cambiarme de branch
git merge "name_branch" preferiblemente hacerlo desde master para reflejar los cambios
git branch -d "name_branch" para eliminar una branch
git clone "nombre_repositorio""my_name"
git remote -v visualiza un (fetch) y un (push)
git fetch	para cualquier cambio que haya realizado en mi clon
git fetch origin 	actualizo mis cambios en mi repositorio clon
git merge origin/master 	actualizo en mi repositorio original
