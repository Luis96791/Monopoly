    sf::Vector2f mouse = window5.mapPixelToCoords(sf::Mouse::getPosition(window5));
    sf::FloatRect bounds = a.getGlobalBounds();

    if(bounds.contains(mouse)){
        if(sf::Mouse::isButtonPressed(sf::Mouse::Left)){
            return true;
        }
    }
